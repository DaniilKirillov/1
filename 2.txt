#include <iostream> 
#include <vector> 
using namespace std; 
int prime(int a) { 
	for (int i = 2; i <= a/2; i++) { 
		if (a % i == 0) return 0; 
		return 1; 
	} 
} 
void delete_annoying_primes(int n, int *a) { 
	for (int i = 0; i < n; i++) { 
		cin >> a[i]; 
	} 
	for (int i1 = 0; i1<n; i1++) { 
		int l = 0; 
		for (int i2 = 0; i2<n; i2++) { 
			if (a[i1] == a[i2]) l++; 
		} 
		if (l == 1 || a[i1] == 0 || prime(a[i1]) == 0) cout « a[i1] « " "; 
	} 
} 
int main () { 
	int n; 
	cin >> n; 
	int *a = new int [n];
	delete_annoying_primes(n, a); 
	return 0; 
}
